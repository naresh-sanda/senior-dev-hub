Author: Naresh Kumar Sanda

The fundamentals of Python programming in one place.

1. Data Types & Structures

# Basic Types
x = 10          # int
y = 3.14        # float
name = "Alice"  # str
is_active = True # bool

# Lists (Mutable, Ordered)
fruits = ["apple", "banana", "cherry"]
fruits.append("orange")
print(fruits[0]) # "apple"

# Tuples (Immutable, Ordered)
coordinates = (10, 20)

# Sets (Mutable, Unordered, Unique elements)
unique_numbers = {1, 2, 3, 3} # Becomes {1, 2, 3}

# Dictionaries (Key-Value pairs)
person = {
    "name": "Bob",
    "age": 25
}
print(person["name"]) # "Bob"

2. Control Flow

# If-Elif-Else
age = 20
if age < 18:
    print("Minor")
elif age == 18:
    print("Just became an adult")
else:
    print("Adult")

# For Loops
for item in fruits:
    print(item)

# For Loop with range (start, stop, step)
for i in range(0, 10, 2):
    print(i) # Prints 0, 2, 4, 6, 8

# While Loops
count = 0
while count < 5:
    print(count)
    count += 1

3. Functions & Scope

# Defining a function
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}!"

print(greet("Alice")) # "Hello, Alice!"
print(greet("Bob", "Hi")) # "Hi, Bob!"

# *args for variable positional arguments
def sum_all(*args):
    return sum(args)

# **kwargs for variable keyword arguments
def print_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_info(name="Charlie", age=30)

4. Object-Oriented Programming

class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        return "Some generic sound"

# Inheritance
class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name) # Call parent constructor
        self.breed = breed

    # Method Overriding
    def speak(self):
        return "Woof!"

my_dog = Dog("Rex", "German Shepherd")
print(my_dog.speak()) # "Woof!"

5. Exception Handling

try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: Cannot divide by zero. ({e})")
except Exception as e:
    print(f"A general error occurred: {e}")
finally:
    print("This runs no matter what.")

6. File I/O

Always use the with statement so the file is automatically closed, even if an exception occurs.

# Writing to a file
with open("data.txt", "w") as file:
    file.write("Hello, World!\n")
    file.write("Python is great.")

# Reading from a file
with open("data.txt", "r") as file:
    contents = file.read()
    print(contents)
