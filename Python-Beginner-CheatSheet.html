1. Variables & Types

# Python is dynamically typed
name = "Alice"       # string
age = 30             # integer
height = 5.6         # float
is_developer = True  # boolean

# Check type
print(type(age))     # <class 'int'>

2. Control Flow

# If-Else Statement
if age >= 18:
    print("Adult")
elif age > 12:
    print("Teen")
else:
    print("Child")

# For Loop
for i in range(5):
    print(i) # 0, 1, 2, 3, 4

# While Loop
count = 0
while count < 3:
    print(count)
    count += 1

3. Data Structures

# Lists (Mutable)
fruits = ["apple", "banana", "cherry"]
fruits.append("orange")
print(fruits[0]) # "apple"

# Dictionaries (Key-Value)
person = {"name": "Bob", "age": 25}
person["city"] = "New York"
print(person["name"]) # "Bob"

# Sets (Unique items)
unique_numbers = {1, 2, 3, 3, 2}
print(unique_numbers) # {1, 2, 3}

# Tuples (Immutable)
coordinates = (10, 20)

4. Functions

# Define a function
def greet(name):
    return f"Hello, {name}!"

# Call the function
message = greet("Alice")
print(message) # "Hello, Alice!"

# Default parameters
def power(base, exp=2):
    return base ** exp

print(power(3))    # 9 (3^2)
print(power(3, 3)) # 27 (3^3)
